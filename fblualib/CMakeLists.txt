CMAKE_MINIMUM_REQUIRED(VERSION 2.8.7 FATAL_ERROR)

# Torch messes this up
SET(SAVED_CMAKE_INSTALL_PREFIX ${CMAKE_INSTALL_PREFIX})

# Use our own cmake scripts
SET(CMAKE_MODULE_PATH
  "${CMAKE_CURRENT_SOURCE_DIR}/cmake"
  "${CMAKE_MODULE_PATH}")

INCLUDE(MultiLevelIncludes)
MLI_SET_DEPTH(1)

FIND_PACKAGE(Folly REQUIRED)
FIND_PACKAGE(Glog REQUIRED)
FIND_PACKAGE(Torch REQUIRED)
FIND_PACKAGE(THPP REQUIRED)

INCLUDE_DIRECTORIES(
  ${FOLLY_INCLUDE_DIR}
  ${GLOG_INCLUDE_DIR}
  ${THPP_INCLUDE_DIR}
)

SET(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -std=gnu++11")

SET(src
  LuaUtils.cpp
)

SET(h
  LuaUtils.h
  LuaUtils-inl.h
)

SET(CMAKE_SHARED_LINKER_FLAGS "-Wl,--no-as-needed")
ADD_LIBRARY(fblualib SHARED ${src})

TARGET_LINK_LIBRARIES(fblualib
  luaT ${THPP_LIBRARIES} ${FOLLY_LIBRARIES} ${GLOG_LIBRARIES})

SET(CMAKE_INSTALL_PREFIX ${SAVED_CMAKE_INSTALL_PREFIX})

INSTALL(TARGETS fblualib
        LIBRARY DESTINATION lib)

INSTALL(FILES ${h}
        DESTINATION "include/${MLI_INCLUDE_RELATIVE_OUTPUT_DIR}")
